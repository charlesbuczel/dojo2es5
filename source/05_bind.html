Function.prototype.bind()

            
require(["dojo/_base/lang"], function(lang){
    var myObj = { foo: "bar" },
        func = lang.hitch(myObj, function(){
            console.log(this.foo);
        });

        func();
});
becomes: 

            
var myObj = { foo: "bar" },
    func = function(){
        console.log(this.foo);
    }.bind(myObj);

func();
